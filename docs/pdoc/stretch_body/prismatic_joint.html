 98    def get_soft_motion_limits(self):
 99        """
100            Return the currently applied soft motion limits: [min, max]
101
102            The soft motion limit restricts joint motion to be <= its physical limits.
103
104            There are three types of limits:
105            Hard: The physical limits
106            Collision: Limits set by RobotCollision to avoid collisions
107            User: Limits set by the user software
108
109            The joint is limited to the most restrictive range of the Hard / Collision/ User values.
110            Specifying a value of None for a limit indicates that no constraint exists for that limit type.
111            This allows a User limits and Collision limits to coexist.
112            For example, a user can temporarily restrict the range of motion beyond the current collision limits.
113            Then, by commanding User limits of None, the joint limits will revert back to the collision settings.
114        """
115        return self.soft_motion_limits['current']

145    def contact_thresh_to_motor_current(self,contact_thresh_pos, contact_thresh_neg):
146        """
147        This model converts from a specified percentage effort (-100 to 100) in the motor frame to motor currents
148        """
149        e_cn = self.params['contact_models']['effort_pct']['contact_thresh_default'][0] if contact_thresh_neg is None else contact_thresh_neg
150        e_cp = self.params['contact_models']['effort_pct']['contact_thresh_default'][1] if contact_thresh_pos is None else contact_thresh_pos
151        i_contact_neg = self.motor.effort_pct_to_current(max(e_cn, self.params['contact_models']['effort_pct']['contact_thresh_max'][0]))
152        i_contact_pos = self.motor.effort_pct_to_current(min(e_cp, self.params['contact_models']['effort_pct']['contact_thresh_max'][1]))
153        return i_contact_pos, i_contact_neg

208    def _step_vel_braking(self, v_des, a_des, stiffness, i_feedforward, i_contact_pos, i_contact_neg):
209        """
210        In velocity mode while using set_velocity() command, when the joint is in a braking zone,
211        the input velocities are tapered till the joint limits  to zero and smoothly braked at the limits to 
212        avoid hitting the hardstops.
213        """
214        if self._prev_set_vel_ts is None:
215            self._prev_set_vel_ts = time.time()
216        if self.status['timestamp_pc']>self._prev_set_vel_ts: # Braking control syncs with the pull status's freaquency for accurate motion control
217            # Honor joint limits in velocity mode
218            lim_lower = self.get_soft_motion_limits()[0]
219            lim_upper = self.get_soft_motion_limits()[1]
220            
221            v_curr = self.status['vel']
222            x_curr = self.status['pos']
223
224            to_min = abs(x_curr - lim_lower)
225            to_max = abs(x_curr - lim_upper)
226
227            c1 = to_min<to_max and v_des>0 # if v_des -ve
228            c2 = to_min>to_max and v_des<0 # if v_des +ve
229            opp_vel = c1 or c2 
230            
231            v = 0
232            if opp_vel:
233                v = v_des # allow input velocity if direction is opposite to nearest limit
234            elif min(to_max,to_max)<0.001:
235                v = 0  # apply brakes if the braking distance is >= limits
236            else:
237                taper = min(to_max,to_min)/self.vel_brake_zone_thresh # normalized (0~1) distance to limits
238                v = v_des*taper # apply tapered velocity inside braking zone
239            
240            # convert to motor rad
241            v_m=min(self.params['motion']['max']['vel_m'],v) if v>=0 else max(-1*self.params['motion']['max']['vel_m'],v)
242            v_r = self.translate_m_to_motor_rad(v_m)
243
244            self.logger.debug(f"Applied safety brakes near limits. reduced set_vel={v_m} m/s")
245            self.motor.set_command(mode=Stepper.MODE_VEL_TRAJ,
246                                v_des=v_r,
247                                a_des=a_des,
248                                stiffness=stiffness,
249                                i_feedforward=i_feedforward,
250                                i_contact_pos=i_contact_pos,
251                                i_contact_neg=i_contact_neg)
252            self._prev_set_vel_ts = time.time()

369    def follow_trajectory(self, v_m=None, a_m=None, stiffness=None, contact_thresh_pos_N=None,contact_thresh_neg_N=None,
370                          req_calibration=True, move_to_start_point=True,contact_thresh_pos=None,contact_thresh_neg=None):
371        """Starts executing a waypoint trajectory
372
373        `self.trajectory` must be populated with a valid trajectory before calling
374        this method.
375
376        Parameters
377        ----------
378        v_m : float
379            velocity limit for trajectory in meters per second
380        a_m : float
381            acceleration limit for trajectory in meters per second squared
382        stiffness: stiffness of motion. Range 0.0 (min) to 1.0 (max)
383        contact_thresh_pos_N: Deprecated: positive threshold to stop motion (units: pseudo_N)
384        contact_thresh_neg_N: Deprecated: negative threshold to stop motion (units: pseudo_N)
385        contact_thresh_pos: positive threshold to stop motion (units: effort_pct -100:100)
386        contact_thresh_neg: negative threshold to stop motion (units: effort_pct -100:100)
387        req_calibration : bool
388            whether to allow motion prior to homing
389        move_to_start_point : bool
390            whether to move to the trajectory's start to avoid a jump, this
391            time to move doesn't count against the trajectory's timeline
392        """
393
394
395        hu.check_deprecated_contact_model_prismatic_joint(self,'follow_trajectory', contact_thresh_pos_N, contact_thresh_neg_N,contact_thresh_pos, contact_thresh_neg)
396
397        # check if joint valid, homed, and right protocol
398        if not self.motor.hw_valid:
399            self.logger.warning('%s connection to hardware not valid'%self.name.upper)
400            return False
401        if req_calibration:
402            if not self.motor.status['pos_calibrated']:
403                self.logger.warning('%s not calibrated'%self.name.capitalize())
404                return False
405        if int(str(self.motor.board_info['protocol_version'])[1:]) < 1:
406            self.logger.warning("%s firmware version doesn't support waypoint trajectories"%self.name.capitalize())
407            return False
408
409        # check if trajectory valid
410        vel_limit = v_m if v_m is not None else self.params['motion']['trajectory_max']['vel_m']
411        acc_limit = a_m if a_m is not None else self.params['motion']['trajectory_max']['accel_m']
412        valid, reason = self.trajectory.is_valid(vel_limit, acc_limit)
413        if not valid:
414            self.logger.warning('Joint traj not valid: {0}'.format(reason))
415            return False
416        if valid and reason == "must have at least two waypoints":
417            # skip this device
418            return True
419
420        # set defaults
421        stiffness = max(0.0, min(1.0, stiffness)) if stiffness is not None else self.stiffness
422        v_r = self.translate_m_to_motor_rad(min(abs(v_m), self.params['motion']['trajectory_max']['vel_m'])) \
423            if v_m is not None else self.translate_m_to_motor_rad(self.params['motion']['trajectory_max']['vel_m'])
424        a_r = self.translate_m_to_motor_rad(min(abs(a_m), self.params['motion']['trajectory_max']['accel_m'])) \
425            if a_m is not None else self.translate_m_to_motor_rad(self.params['motion']['trajectory_max']['accel_m'])
426
427        i_contact_pos,i_contact_neg  = self.contact_thresh_to_motor_current(contact_thresh_pos,contact_thresh_neg )
428
429
430        # move to start point
431        if move_to_start_point:
432            prev_sync_mode = self.motor.gains['enable_sync_mode']
433            self.move_to(self.trajectory[0].position)
434            self.motor.disable_sync_mode()
435            self.push_command()
436            if not self.motor.wait_until_at_setpoint():
437                self.logger.warning('%f unable to reach starting point'%self.name.capitalize())
438            if prev_sync_mode:
439                self.motor.enable_sync_mode()
440                self.push_command()
441            #print('WAIT')
442            #time.sleep(2.0)
443
444        self.traj_guarded_event=self.motor.status['guarded_event']
445        # start trajectory
446        self.motor.set_command(mode=Stepper.MODE_POS_TRAJ_WAYPOINT,
447                               v_des=v_r,
448                               a_des=a_r,
449                               stiffness=stiffness,
450                               i_feedforward=self.i_feedforward,

 99    def get_soft_motion_limits(self):
100        """
101            Return the currently applied soft motion limits: [min, max]
102
103            The soft motion limit restricts joint motion to be <= its physical limits.
104
105            There are three types of limits:
106            Hard: The physical limits
107            Collision: Limits set by RobotCollision to avoid collisions
108            User: Limits set by the user software
109
110            The joint is limited to the most restrictive range of the Hard / Collision/ User values.
111            Specifying a value of None for a limit indicates that no constraint exists for that limit type.
112            This allows a User limits and Collision limits to coexist.

114            Then, by commanding User limits of None, the joint limits will revert back to the collision settings.
115        """
116        return self.soft_motion_limits['current']

209    def _step_vel_braking(self, v_des, a_des, stiffness, i_feedforward, i_contact_pos, i_contact_neg):
210        """
211        In velocity mode while using set_velocity() command, when the joint is in a braking zone,
212        the input velocities are tapered till the joint limits  to zero and smoothly braked at the limits to 
213        avoid hitting the hardstops.
214        """
215        if self._prev_set_vel_ts is None:
216            self._prev_set_vel_ts = time.time()
217        if self.status['timestamp_pc']>self._prev_set_vel_ts: # Braking control syncs with the pull status's freaquency for accurate motion control
218            # Honor joint limits in velocity mode
219            lim_lower = self.get_soft_motion_limits()[0]
220            lim_upper = self.get_soft_motion_limits()[1]
221            
222            v_curr = self.status['vel']
223            x_curr = self.status['pos']
224
225            to_min = abs(x_curr - lim_lower)
226            to_max = abs(x_curr - lim_upper)
227
228            c1 = to_min<to_max and v_des>0 # if v_des -ve
229            c2 = to_min>to_max and v_des<0 # if v_des +ve
230            opp_vel = c1 or c2 
231            
232            v = 0
233            if opp_vel:
234                v = v_des # allow input velocity if direction is opposite to nearest limit
235            elif min(to_max,to_max)<0.001:
236                v = 0  # apply brakes if the braking distance is >= limits
237            else:
238                taper = min(to_max,to_min)/self.vel_brake_zone_thresh # normalized (0~1) distance to limits
239                v = v_des*taper # apply tapered velocity inside braking zone
240            
241            # convert to motor rad
242            v_m=min(self.params['motion']['max']['vel_m'],v) if v>=0 else max(-1*self.params['motion']['max']['vel_m'],v)
243            v_r = self.translate_m_to_motor_rad(v_m)
244
245            self.logger.debug(f"Applied safety brakes near limits. reduced set_vel={v_m} m/s")
246            self.motor.set_command(mode=Stepper.MODE_VEL_TRAJ,
247                                v_des=v_r,
248                                a_des=a_des,
249                                stiffness=stiffness,
250                                i_feedforward=i_feedforward,
251                                i_contact_pos=i_contact_pos,
252                                i_contact_neg=i_contact_neg)
253            self._prev_set_vel_ts = time.time()

This allows a User limits and Collision limits to coexist. For example, a user can temporarily restrict the range of motion beyond the current collision limits. Then, by commanding User limits of None, the joint limits will revert back to the collision settings.
