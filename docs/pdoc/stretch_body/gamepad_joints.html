  4"""
  5The gamepad_joints library provides the abstract motion command classes 
  6for each robot joint that can be used in a control loop to make a motion through an gamepad 
  7type inputs elements (Button presses, Analog Stick motions).
  8
  9The Gamepad joints command classes primarily uses velocity controls. All the 
 10acceleration profiles are dynamically optimized based on the user input type to 
 11provide smooth and responsive robot motions.
 12
 13A gamepad joint command class will provide the below four main attributs 
 14to convert a gamepad input to an appropriate motion:
 15
 16command_stick_to_motion()
 17    Supply a float value between -1.0 to 1.0 from a control loop. 
 18    The value supplied and it's sign determines the speed of joint motion and direction
 19    Use this method to map values from an analog UI elements to a joint motion.
 20    Note the base motion class needs an aditional y axis value / x,y axis values for  linear,rotion motion.
 21
 22command_button_to_motion()
 23    Supply a direction integere either +1 or -1 in a control loop for the joint to move in that direction
 24    Use this method to map a boolean button state UI elements to a joint motion.
 25    
 26stop_motion()
 27    Use this method when ever a joints needs to be still with no motion in a control loop.
 28
 29precision_mode
 30    Set this flag to true or false to enable and disable precision mode for each joint.
 31
 32fast_base_mode
 33    This flag is specific to base command class. Set this flag True to move the base
 34    faster but only allowed when the arm and lift are in safe positions to avoid tipping.
 35"""

 37class CommandBase:
 38    def __init__(self):
 39        """Base motion command class
 40        """
 41        self.params = RobotParams().get_params()[1]['base']
 42        self.dead_zone = 0.0001
 43        self._prev_set_vel_ts = None
 44        self.max_linear_vel = self.params['motion']['max']['vel_m']
 45        self.max_rotation_vel = 1.90241 # rad/s
 46        self.normal_linear_vel = self.params['motion']['default']['vel_m']
 47        self.normal_rotation_vel = self.max_rotation_vel*0.4
 48        self.arm_extended_rotation_vel = self.max_rotation_vel*0.2
 49        self.precision_mode = False 
 50        self.fast_base_mode = False
 51        self.acc = self.params['motion']['max']['accel_m']
 52
 53        # Precision mode params
 54        self.precision_max_linear_vel = 0.05 #0.02 m/s Very precise: 0.01
 55        self.precision_max_rot_vel = 0.3 #0.08 rad/s Very precise: 0.04

 57    def command_stick_to_motion(self, x, y, robot):
 58        """Convert a stick axis value to robot base's tank driving motion.
 59
 60        Args:
 61            x (float): Range [-1.0,+1.0], control rotation speed
 62            y (float): Range [-1.0,+1.0], control linear speed
 63            robot (robot.Robot): Valid robot instance
 64        """
 65        if abs(x) < self.dead_zone:
 66            x = 0
 67        if abs(y) < self.dead_zone:
 68            y = 0
 69        x = to_parabola_transform(x)
 70        # y = to_parabola_transform(y) 
 71        
 72        # Standard Mode
 73        if not self.precision_mode:
 74            self.start_pos = None
 75            self.start_theta = None
 76            v_m, w_r = self._process_stick_to_vel(x,y, robot)
 77            robot.base.set_velocity(v_m, w_r, a=self.acc)
 78            self._prev_set_vel_ts = time.time()
 79        else:
 80        # Precision Mode
 81            if self.start_pos is None:
 82                self.start_pos = [robot.base.status['x'],robot.base.status['y']]
 83                self.start_theta = robot.base.status['theta']
 84                self.target_position = self.start_pos
 85                self.target_theta = self.start_theta  
 86            yv = map_to_range(abs(y),0,self.precision_max_linear_vel)
 87            xv = map_to_range(abs(x),0,self.precision_max_rot_vel)
 88            if x<0:
 89                xv = -1*xv
 90            if y<0:
 91                yv = -1*yv
 92            if abs(x)>abs(y):
 93                self._step_precision_rotate(xv, robot)
 94            else:
 95                self._step_precision_translate(yv, robot)
 96            # Update the previous_time for the next iteration
 97            self._prev_set_vel_ts = time.time()
 98    
 99    def command_button_to_rotation_motion(self, direction, robot):
100        """Make robot base rotation motion based on a button state.
101
102        Args:
103            direction (int): Direction integer -1 or +1
104            robot (robot.Robot): Valid robot instance
105        """
106        v_m = 0
107        w_r = direction*self.normal_rotation_vel
108        if self.fast_base_mode and self.is_fastbase_safe(robot):
109            w_r = direction*self.max_rotation_vel
110        if self.precision_mode:
111            w_r = direction*self.precision_max_rot_vel
112        robot.base.set_velocity(v_m, w_r, a=self.acc)
113        # Update the previous_time for the next iteration
114        self._prev_set_vel_ts = time.time()

142    def is_fastbase_safe(self, robot):
143        """Check if the base is fast navigation mode safe
144
145        Args:
146            robot (robot.Robot): Valid robot instance
147
148        Returns:
149            bool: True if fast navigation safe
150        """
151        arm = robot.arm.status['pos'] < (robot.get_stow_pos('arm') + 0.1) # check if arm pos under stow pose + 0.1 m
152        lift = robot.lift.status['pos'] < (robot.get_stow_pos('lift') + 0.05) # check if lift pos is under stow pose + 0.05m
153        return arm and lift

155    def _process_stick_to_vel(self, x, y, robot):
156        max_linear_vel = self.normal_linear_vel
157        max_rotation_vel = self.normal_rotation_vel
158        if self.fast_base_mode and self.is_fastbase_safe(robot):
159            max_linear_vel =  self.max_linear_vel
160            max_rotation_vel = self.max_rotation_vel
161        if not robot.arm.status['pos'] < (robot.get_stow_pos('arm') + 0.1):
162            max_rotation_vel = self.arm_extended_rotation_vel
163        v_m = map_to_range(abs(y), 0, max_linear_vel)
164        if y<0:
165            v_m = -1*v_m
166        x = -1*x
167        w_r = map_to_range(abs(x), 0, max_rotation_vel)
168        if x<0:
169            w_r = -1*w_r
170        return v_m, w_r

172    def _step_precision_rotate(self, xv, robot):
173        # Calculate the time elapsed since the last iteration
174        current_time = time.time()
175        elapsed_time = current_time - self._prev_set_vel_ts 
176
177        # Calculate the desired change in position to achieve the desired velocity
178        desired_theta_change = xv * elapsed_time
179
180        # Set the control_effort as the position setpoint for the joint
181        if abs(xv)>=0:
182            robot.base.rotate_by(-1*desired_theta_change)
183
184    def _step_precision_translate(self, yv, robot):
185        
186        # Calculate the time elapsed since the last iteration
187        current_time = time.time()
188        elapsed_time = current_time - self._prev_set_vel_ts 
189
190        # Calculate the desired change in position to achieve the desired velocity
191        desired_position_change = yv * elapsed_time
192
193        # Set the control_effort as the position setpoint for the joint
194        if abs(yv)>=0:
195            robot.base.translate_by(desired_position_change)

412class CommandDxlJoint:
413    """Abstract motion command class for Dynamixel joints
414    """
415    def __init__(self, name, max_vel=None, acc_type=None):
416        """Initiate a Dynamixe joint either a head_* or wrist_* group.
417
418        Args:
419            name (str): Name of the device name
420            max_vel (float, optional): Set a custom max velocity (rad/s)
421            acc_type (str, optional): Set custom acceleration profile (fast,slow,default)
422        """
423        self.params = RobotParams().get_params()[1][name]
424        self.name = name
425        self.dead_zone = 0.001
426        self._prev_set_vel_ts = None
427        self.max_vel = max_vel if max_vel else self.params['motion']['default']['vel']
428        self.precision_mode = False
429        self.acc = None
430        if acc_type:
431            self.acc = self.params['motion'][acc_type]['accel']
432        
433        self.precision_scale_down = 0.05
434
435    def command_stick_to_motion(self, x, robot):
436        """Convert a stick axis value to dynamixel servo motion.
437
438        Args:
439            x (float): Range [-1.0,+1.0], control servo speed
440            robot (robot.Robot): Valid robot instance
441        """
442        if 'wrist' in self.name:
443            motor = robot.end_of_arm.get_joint(self.name)
444        if 'head' in self.name:
445            motor = robot.head.get_joint(self.name)
446        if abs(x)<self.dead_zone:
447            x = 0
448        acc = self.acc
449        if x==0:
450            acc = self.params['motion']['max']['accel'] #Stop with Strong Acceleration
451        # x = to_parabola_transform(x)
452        v = self._process_stick_to_vel(x)
453        if self.precision_mode:
454            v = v*self.precision_scale_down
455        motor.set_velocity(v, acc)
456        self._prev_set_vel_ts = time.time()
457
458    def command_button_to_motion(self,direction, robot):
459        """Make servo move based on a button state.
460
461        Args:
462            direction (int): Direction integer -1 or +1
463            robot (robot.Robot): Valid robot instance
464        """
465        if 'wrist' in self.name:
466            motor = robot.end_of_arm.get_joint(self.name)
467        if 'head' in self.name:
468            motor = robot.head.get_joint(self.name)
469        vel = self.max_vel
470        if self.precision_mode:
471            vel = vel*self.precision_scale_down
472        if direction==1:
473            motor.set_velocity(vel, self.acc)
474        elif direction==-1:
475            motor.set_velocity(-1*vel, self.acc)
476        self._prev_set_vel_ts = time.time()
477    
478    def stop_motion(self, robot):
479        """Stop the joint motion. To be used when ever the controller is idle/no-inputs
480        to stop unnecessary robot motion.
481
482        Args:
483            robot (robot.Robot): Valid robot instance
484        """
485        if 'wrist' in self.name:
486            motor = robot.end_of_arm.get_joint(self.name)
487        if 'head' in self.name:
488            motor = robot.head.get_joint(self.name)
489        
490        motor.set_velocity(0,self.params['motion']['max']['accel'])
