 76class NonDXLStatusThread(threading.Thread):
 77    """
 78    This thread runs at 25Hz.
 79    It updates the status data of the Devices.
 80    It also steps the Sentry, Monitor, and Collision functions
 81    """
 82    def __init__(self, robot, target_rate_hz=25.0):
 83        threading.Thread.__init__(self, name = self.__class__.__name__)
 84        self.robot=robot
 85        self.robot_update_rate_hz = target_rate_hz
 86        self.shutdown_flag = threading.Event()
 87        self.stats = hello_utils.LoopStats(loop_name='NonDXLStatusThread',target_loop_rate=self.robot_update_rate_hz)
 88        self.titr=0
 89        self.first_status = False
 90        self.loop = asyncio.new_event_loop()
 91        self.running = False
 92
 93    def step(self):
 94        asyncio.set_event_loop(self.loop)
 95        self.stats.mark_loop_start()
 96        if self.robot.params['use_asyncio']:
 97            asyncio.get_event_loop().run_until_complete(asyncio.gather(
 98                self.robot.wacc.pull_status_async(),
 99                self.robot.base.pull_status_async(),
100                self.robot.lift.pull_status_async(),
101                self.robot.arm.pull_status_async(),
102                self.robot.pimu.pull_status_async()))
103        else:
104            self.robot._pull_status_non_dynamixel()
105        self.stats.mark_loop_end()
106    def run(self):
107        self.running=True
108        while not self.shutdown_flag.is_set():
109            self.stats.wait_until_ready_to_run()
110            if not self.shutdown_flag.is_set():
111                self.step()
112            self.first_status = True
113        self.robot.logger.debug('Shutting down NonDXLStatusThread')

115class SystemMonitorThread(threading.Thread):
116    """
117    This thread runs at 25Hz.
118    It updates the status data of the Devices.
119    It also steps the Sentry, Monitor, and Collision functions
120    """
121    def __init__(self, robot, target_rate_hz=25.0):
122        threading.Thread.__init__(self, name = self.__class__.__name__)
123        self.robot=robot
124        self.robot_update_rate_hz = target_rate_hz
125        self.monitor_downrate_int = int(robot.params['rates']['SystemMonitorThread_monitor_downrate_int'])  # Step the monitor at every Nth iteration
126        self.trace_downrate_int = int(robot.params['rates']['SystemMonitorThread_trace_downrate_int'])  # Step the trace at every Nth iteration
127        self.sentry_downrate_int = int(robot.params['rates']['SystemMonitorThread_sentry_downrate_int']) # Step the sentry at every Nth iteration
128        self.collision_downrate_int = int(robot.params['rates']['SystemMonitorThread_collision_downrate_int'])  # Step the monitor at every Nth iteration
129        self.trajectory_downrate_int = int(robot.params['rates']['SystemMonitorThread_nondxl_trajectory_downrate_int'])  # Update hardware with waypoint trajectory segments at every Nth iteration
130
131        if self.robot.params['use_collision_manager']:
132            self.robot.collision.startup()
133        if self.robot.params['use_monitor']:
134            self.robot.monitor.startup()
135        self.shutdown_flag = threading.Event()
136        self.stats = hello_utils.LoopStats(loop_name='SystemMonitorThread',target_loop_rate=self.robot_update_rate_hz)
137        self.titr=0
138        self.running=False
139
140    def step(self):
141        self.titr = self.titr + 1
142        self.stats.mark_loop_start()
143        if self.robot.params['use_monitor']:
144            if (self.titr % self.monitor_downrate_int) == 0:
145                self.robot.monitor.step()
146        if self.robot.params['use_trace']:
147            if (self.titr % self.trace_downrate_int) == 0:
148                self.robot.trace.step()
149        if self.robot.params['use_sentry']:
150            if (self.titr % self.sentry_downrate_int) == 0:
151                self.robot._step_sentry()
152        if self.robot.params['use_collision_manager'] and self.robot.is_calibrated():
153            self.robot.collision.step()
154        if (self.titr % self.trajectory_downrate_int) == 0:
155            self.robot._update_trajectory_non_dynamixel()
156        self.stats.mark_loop_end()
157
158    def run(self):
159        self.running=True
160        while not self.shutdown_flag.is_set():
161            self.stats.wait_until_ready_to_run()
162            if not self.shutdown_flag.is_set():
163                self.step()
164        self.robot.logger.debug('Shutting down SystemMonitorThread')

212    def startup(self,start_non_dxl_thread=True,start_dxl_thread=True,start_sys_mon_thread=True):
213        """
214        To be called once after class instantiation.
215        Prepares devices for communications and motion
216
217        Returns
218        -------
219        bool
220            true if startup of robot succeeded
221        """
222        self.logger.debug('Starting up Robot {0} of batch {1}'.format(self.params['serial_no'], self.params['batch_name']))
223        success = True
224        for k in self.devices:
225            if self.devices[k] is not None:
226                if not self.devices[k].startup(threaded=False):
227                    success = False
228
229        if (self.arm.motor.transport.version==0 \
230                or self.lift.motor.transport.version==0 \
231                or self.base.right_wheel.transport.version == 0 \
232                or self.base.left_wheel.transport.version == 0 \
233                or self.wacc.transport.version==0 \
234                or self.pimu.transport.version==0) and self.params['use_asyncio'] :
235            self.logger.warning('Not able to use asyncio for transport communications. Defaulting to sync.')
236            self.params['use_asyncio']=0
237        else:
238            self.start_event_loop()
239
240        # Register the signal handlers
241        signal.signal(signal.SIGTERM, hello_utils.thread_service_shutdown)
242        signal.signal(signal.SIGINT, hello_utils.thread_service_shutdown)
243
244        self.non_dxl_thread = NonDXLStatusThread(self, target_rate_hz=self.params['rates']['NonDXLStatusThread_Hz'])
245        self.dxl_end_of_arm_thread = DXLEndOfArmStatusThread(self,target_rate_hz=self.params['rates']['DXLStatusThread_Hz'])
246        self.sys_thread = SystemMonitorThread(self, target_rate_hz=self.params['rates']['SystemMonitorThread_Hz'])
247        self.dxl_head_thread = DXLHeadStatusThread(self, target_rate_hz=self.params['rates']['DXLStatusThread_Hz'])
248
249        if start_non_dxl_thread:
250            self.non_dxl_thread.setDaemon(True)
251            self.non_dxl_thread.start()
252            ts = time.time()
253            while not self.non_dxl_thread.first_status and time.time() - ts < 3.0:
254                time.sleep(0.01)
255
256        if start_dxl_thread:
257            self.dxl_head_thread.setDaemon(True)
258            self.dxl_head_thread.start()
259            self.dxl_end_of_arm_thread.setDaemon(True)
260            self.dxl_end_of_arm_thread.start()
261
262        if start_sys_mon_thread:
263            self.sys_thread.setDaemon(True)
264            self.sys_thread.start()
265
266        return success

268    def stop(self):
269        """
270        To be called once before exiting a program
271        Cleanly stops down motion and communication
272        """
273        self.logger.debug('---- Shutting down robot ----')
274        if self.non_dxl_thread.running:
275            self.non_dxl_thread.shutdown_flag.set()
276            self.non_dxl_thread.join(1)
277        if self.dxl_head_thread.running:
278            self.dxl_head_thread.shutdown_flag.set()
279            self.dxl_head_thread.join(1)
280        if self.dxl_end_of_arm_thread.running:
281            self.dxl_end_of_arm_thread.shutdown_flag.set()
282            self.dxl_end_of_arm_thread.join(1)
283        if self.sys_thread.running:
284            self.sys_thread.shutdown_flag.set()
285            self.sys_thread.join(1)
286        for k in self.devices:
287            if self.devices[k] is not None:
288                self.logger.debug('Shutting down %s'%k)
289                self.devices[k].stop()
290        self.stop_event_loop()
291        self.logger.debug('---- Shutdown complete ----')

319    def push_command(self):
320        """
321        Cause all queued up RPC commands to be sent down to Devices
322        """
323        
324        with self.lock:
325            ready = self.pimu.is_ready_for_sync()
326            if self.params['use_asyncio'] and self.async_event_loop is not None:
327                future = asyncio.run_coroutine_threadsafe(self.push_command_coro(),self.async_event_loop)
328                future.result()
329            else:
330                self.base.push_command()
331                self.arm.push_command()
332                self.lift.push_command()
333                self.pimu.push_command()
334                self.wacc.push_command()
335
336            # Check if need to do a motor sync by looking at if there's been a pimu sync signal sent
337            # since the last stepper.set_command for each joint
338            sync_required = (self.pimu.ts_last_motor_sync is not None) and (
339                    self.arm.motor.is_sync_required(self.pimu.ts_last_motor_sync)
340                    or self.lift.motor.is_sync_required(self.pimu.ts_last_motor_sync)
341                    or self.base.right_wheel.is_sync_required(self.pimu.ts_last_motor_sync)
342                        or self.base.left_wheel.is_sync_required(self.pimu.ts_last_motor_sync))
343
344            if (self.pimu.ts_last_motor_sync is None or ( ready and sync_required)):
345                self.pimu.trigger_motor_sync()

356    def follow_trajectory(self):
357        success = True
358        success = success and self.arm.follow_trajectory(move_to_start_point=False)
359        success = success and self.lift.follow_trajectory(move_to_start_point=False)
360        success = success and self.base.follow_trajectory()
361
362        # Check if need to do a motor sync by looking at if there's been a pimu sync signal sent
363        # since the last stepper.set_command for each joint
364        sync_required = (self.pimu.ts_last_motor_sync is not None) and (
365                self.arm.motor.ts_last_syncd_motion > self.pimu.ts_last_motor_sync
366                or self.lift.motor.ts_last_syncd_motion > self.pimu.ts_last_motor_sync
367                or self.base.right_wheel.ts_last_syncd_motion > self.pimu.ts_last_motor_sync
368                or self.base.left_wheel.ts_last_syncd_motion > self.pimu.ts_last_motor_sync)
369        if self.pimu.ts_last_motor_sync is None or sync_required:
370            self.pimu.trigger_motor_sync()
371
372        success = success and self.end_of_arm.follow_trajectory(move_to_start_point=False)
373        success = success and self.head.follow_trajectory(move_to_start_point=False)
374        return success

385    def is_calibrated(self):
386        """
387        Returns true if homing-calibration has been run all joints that require it
388        """
389        ready = self.lift.motor.status['pos_calibrated']
390        ready = ready and self.arm.motor.status['pos_calibrated']
391        for j in self.end_of_arm.joints:
392            req = self.end_of_arm.motors[j].params['req_calibration'] and not self.end_of_arm.motors[j].is_calibrated
393            ready = ready and not req

406    def stow(self):
407        """
408        Cause the robot to move to its stow position
409        Blocking.
410        """
411        self.head.move_to('head_pan', self.get_stow_pos('head_pan'))
412        self.head.move_to('head_tilt',self.get_stow_pos('head_pan'))
413
414        lift_stowed=False
415        pos_lift = self.get_stow_pos('lift')
416        if self.lift.status['pos']<=pos_lift: #Needs to come up before bring in arm
417            print('--------- Stowing Lift ----')
418            self.lift.move_to(pos_lift)
419            self.push_command()
420            time.sleep(0.25)
421            ts = time.time()
422            while not self.lift.motor.status['near_pos_setpoint'] and time.time() - ts < 4.0:
423                time.sleep(0.1)
424            lift_stowed=True
425
426        #Bring in arm before bring down
427        print('--------- Stowing Arm ----')
428        self.arm.move_to(self.get_stow_pos('arm'))
429        self.push_command()
430        time.sleep(0.25)
431        ts = time.time()
432        while not self.arm.motor.status['near_pos_setpoint'] and time.time() - ts < 6.0:
433            time.sleep(0.1)
434        self.end_of_arm.stow()
435        time.sleep(0.25)
436
437        #Now bring lift down
438        if not lift_stowed:
439            print('--------- Stowing Lift ----')
440            self.lift.move_to(pos_lift)
441            self.push_command()
442            time.sleep(0.25)
443            ts = time.time()
444            while not self.lift.motor.status['near_pos_setpoint'] and time.time() - ts < 12.0:
445                time.sleep(0.1)
446        #Make sure wrist yaw is done before exiting
447        while self.end_of_arm.motors['wrist_yaw'].motor.is_moving():
448            time.sleep(0.1)
