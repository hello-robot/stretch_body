115def overwrite_dict(overwritee_dict, overwriter_dict):
116    """Merge two dictionaries while overwriting common keys and
117    report errors when values of the same key differ in Python
118    type. The result gets stored in `overwritee_dict`.
119
120    Parameters
121    ----------
122    overwritee_dict : dict
123        The dictionary which will be overwritten. Use this as the merged result.
124    overwriter_dict : dict
125        The dictionary which will overwrite.
126
127    Returns
128    -------
129    bool
130        True if no mismatches were found during the overwrite, False otherwise.
131    """
132    no_mismatches = True
133    for k in overwriter_dict.keys():
134        if k in overwritee_dict:
135            if (isinstance(overwritee_dict[k], dict) and isinstance(overwriter_dict[k], dict)):
136                sub_no_mismatches = overwrite_dict(overwritee_dict[k], overwriter_dict[k])
137                no_mismatches = no_mismatches and sub_no_mismatches
138            else:
139                if (type(overwritee_dict[k]) == type(overwriter_dict[k])) or (isinstance(overwritee_dict[k], numbers.Real) and isinstance(overwriter_dict[k], numbers.Real)):
140                    overwritee_dict[k] = overwriter_dict[k]
141                else:
142                    no_mismatches = False
143                    print('stretch_body.hello_utils.overwrite_dict ERROR: Type mismatch for key={0}, between overwritee={1} and overwriter={2}'.format(k, overwritee_dict[k], overwriter_dict[k]), file=sys.stderr)
144        else: #If key not present, add anyhow (useful for overlaying params)
145            overwritee_dict[k] = overwriter_dict[k]
146    return no_mismatches

244        self.last_ts_loop_start = self.ts_loop_start
245
246        # Calculate sleep time to achieve desired loop rate
247        self.sleep_time_s = (1 / self.target_loop_rate) - self.status['execution_time_s']

296def evaluate_polynomial_at(poly, t):
297    """Evaluate a quintic polynomial at a given time.
298
299    Parameters
300    ----------
301    poly : List(float)
302        Represents a quintic polynomial as a coefficients array [a0, a1, a2, a3, a4, a5].
303        The polynomial is f(t) = a0 + a1*t + a2*t^2 + a3*t^3 + a4*t^4 + a5*t^5
304    t : float
305        the time in seconds at which to evaluate the polynomial
306
307    Returns
308    -------
309    Tuple(float)
310        array with three elements: evaluated position, velocity, and acceleration.
311    """
312    a = [float(elem) for elem in poly]
313    t = float(t)
314    pos = a[0] + (a[1]*t) + (a[2]*t**2) + (a[3]*t**3) + (a[4]*t**4) + (a[5]*t**5)
315    vel = a[1] + (2*a[2]*t) + (3*a[3]*t**2) + (4*a[4]*t**3) + (5*a[5]*t**4)
316    accel = (2*a[2]) + (6*a[3]*t) + (12*a[4]*t**2) + (20*a[5]*t**3)
317    return (pos, vel, accel)

319def is_segment_feasible(segment, v_des, a_des, t=0.0, inc=0.1):
320    """Determine whether a segment adheres to dynamic limits.
321
322    Parameters
323    ----------
324    segment : List
325        Represents a segment of a waypoint trajectory as a list of length eight,
326        structured like [duration_s, a0, a1, a2, a3, a4, a5, segment_id].
327    v_des : float
328        Velocity limit that the segment shouldn't exceed
329    a_des : float
330        Acceleration limit that the segment shouldn't exceed
331    t : float
332        optional, time in seconds at which to begin checking segment
333    inc : float
334        optional, increment in seconds at which the polynomial is evaluated along the segment
335
336    Returns
337    -------
338    success: bool
339        whether the segment is feasible
340    max_v: float
341        Maximum velocity of spline
342    max_a: float
343        Maximum acceleration of spline
344    """
345    v_des = float(v_des)
346    a_des = float(a_des)
347    max_v = 0
348    max_a =0
349    success=True
350    while t < segment[0]:
351        x_t, vel_t, acc_t = evaluate_polynomial_at(segment[1:-1], t)
352        max_v=max(max_v,abs(vel_t))
353        max_a = max(max_a, abs(acc_t))
354        if abs(vel_t) > v_des or abs(acc_t) > a_des:
355            success=False
356        t = min(segment[0], t + inc)
357    return success, max_v, max_a

359def generate_quintic_polynomial(i, f):
360    """Generate quintic polynomial from two points
361
362    Parameters
363    ----------
364    i : List(float)
365        Represents the first waypoint as a list, [time, pos, vel, accel]
366    f : List(float)
367        Represents the second waypoint as a list, [time, pos, vel, accel]
368
369    Returns
370    -------
371    List(float)
372        Represents a quintic polynomial as a coefficients + duration array [duration, a0, a1, a2, a3, a4, a5].
373        The polynomial is f(t) = a0 + a1*t + a2*t^2 + a3*t^3 + a4*t^4 + a5*t^5
374    """
375    i = [float(elem) for elem in i]
376    f = [float(elem) for elem in f]
377    duration = f[0] - i[0]
378    a0 = i[1]
379    a1 = i[2]
380    a2 = i[3] / 2
381    a3 = (20 * f[1] - 20 * i[1] - (8 * f[2] + 12 * i[2]) * duration - (3 * i[3] - f[3]) * (duration ** 2)) / (2 * (duration ** 3))
382    a4 = (30 * i[1] - 30 * f[1] + (14 * f[2] + 16 * i[2]) * duration + (3 * i[3] - 2 * f[3]) * (duration ** 2)) / (2 * (duration ** 4))
383    a5 = (12 * f[1] - 12 * i[1] - (6 * f[2] + 6 * i[2]) * duration - (i[3] - f[3]) * (duration ** 2)) / (2 * (duration ** 5))
384    return [duration, a0, a1, a2, a3, a4, a5]

386def generate_cubic_polynomial(i, f):
387    """Generate cubic polynomial from two points
388
389    Parameters
390    ----------
391    i : List(float)
392        Represents the first waypoint as a list, [time, pos, vel]
393    f : List(float)
394        Represents the second waypoint as a list, [time, pos, vel]
395
396    Returns
397    -------
398    List(float)
399        Represents a cubic polynomial as a coefficients + duration array [duration, a0, a1, a2, a3, 0, 0].
400        The polynomial is f(t) = a0 + a1*t + a2*t^2 + a3*t^3
401    """
402    i = [float(elem) for elem in i]
403    f = [float(elem) for elem in f]
404    duration = f[0] - i[0]
405    a0 = i[1]
406    a1 = i[2]
407    a2 = (3 / duration ** 2) * (f[1] - i[1]) - (2 / duration) * i[2] - (1 / duration) * f[2]
408    a3 = (-2 / duration ** 3) * (f[1] - i[1]) + (1 / duration ** 2) * (f[2] + i[2])
409    return [duration, a0, a1, a2, a3, 0, 0]

434def get_pose_diff(pose0, pose1, translation_atol=2e-3, rotation_atol=2e-2):
435    """Return the motion required to get from pose 0 to pose 1.
436
437    Assumed that between pose 0 and pose 1, there has only been
438    either a translation or rotation motion.
439
440    Parameters
441    ----------
442    pose0 : Tuple(float, float, float)
443        x, y, theta in meters and radians
444    pose1 : Tuple(float, float, float)
445        x, y, theta in meters and radians
446
447    Returns
448    -------
449    float, float
450        Tuple (dx, dtheta) of translation and rotation required to
451        move from pose0 to pose1
452    """
453    x0, y0, theta0 = pose0
454    x1, y1, theta1 = pose1
455    theta0 = np.arctan2(np.sin(theta0), np.cos(theta0)) # constrains to [-pi, pi]
456    theta1 = np.arctan2(np.sin(theta1), np.cos(theta1)) # constrains to [-pi, pi]
457
458    # TODO: For now, we use a simplified motion model where we assume
459    # that every motion is either a translation OR a rotation,
460    # and the translation is either straight forward or straight back
461    if np.isclose(x0, x1, atol=translation_atol) and np.isclose(y0, y1, atol=translation_atol):
462        return 0.0, theta1 - theta0
463
464    if np.isclose(theta0, theta1, atol=rotation_atol):
465        dx = x1 - x0
466        dy = y1 - y0
467        drive_angle = math.atan2(dy, dx)
468        distance = math.hypot(dy, dx)
469        if np.isclose(drive_angle, theta0, atol=rotation_atol):
470            return distance, 0.0
471        opposite_theta0 = np.arctan2(np.sin(theta0 + np.pi), np.cos(theta0 + np.pi)) # constrains to [-pi, pi]
472        if np.isclose(drive_angle, opposite_theta0, atol=rotation_atol):
473            return -distance, 0.0
474
475    return 0.0, 0.0
