301    def add_waypoint(self, new_waypoint):
302        """Add a waypoint to the spline.
303
304        This method will sort through the existing waypoints
305        in the spline to insert the waypoint such that
306        waypoint time increases with index in the array.
307
308        Parameters
309        ----------
310        waypoint : ``Waypoint``
311            with time, position, [velocity, [and acceleration]] attributes set
312        """
313        if len(self.waypoints) == 0:
314            self.waypoints.append(new_waypoint)
315            return
316
317        # Cannot have two waypoints scheduled for the same time
318        if new_waypoint in self.waypoints:
319            return
320
321        # Prepend or append if before first or after last waypoint
322        if new_waypoint < self.waypoints[0]:
323            self.waypoints.insert(0, new_waypoint)
324            return
325        if new_waypoint > self.waypoints[0]:
326            self.waypoints.append(new_waypoint)
327            return
328
329        # Insert before first later waypoint
330        for i, other_waypoint in enumerate(self.waypoints):
331            if new_waypoint < other_waypoint:
332                self.waypoints.insert(i, new_waypoint)
333                return

338    def get_segment(self, index, to_motor_rad=lambda pos: pos):
339        """Retrieves a segment in the spline by index
340
341        Num of segments is one less than number of waypoints in
342        the trajectory. Index bounds are [-1 * num_seg, num_seg).
343
344        Parameters
345        ----------
346        index : int
347            index of segment to return
348        to_motor_rad : func or lambda
349            optional, used to convert waypoint into motor space
350
351        Returns
352        -------
353        Segment
354            coefficients + duration encapsulated in ``Segment`` class
355        """
356        if index < -1 * len(self.waypoints) + 1 or index >= len(self.waypoints) - 1:
357            return None
358
359        index = index - 1 if index < 0 else index
360        w0 = self.waypoints[index].apply_transform(to_motor_rad)
361        w1 = self.waypoints[index + 1].apply_transform(to_motor_rad)
362        return Segment.from_two_waypoints(w0, w1, segment_id=index + 2)

364    def evaluate_at(self, t, to_motor_rad=lambda pos: pos):
365        """Evaluate a point along the curve at a given time.
366
367        Parameters
368        ----------
369        t : float
370            time in seconds
371        to_motor_rad : func or lambda
372            optional, used to convert waypoint into motor space
373
374        Returns
375        -------
376        Tuple(float)
377            tuple with three elements: evaluated position, velocity, and acceleration.
378        """
379        if len(self.waypoints) < 2:
380            return
381
382        # Return bounds for early or late t
383        if t < self.waypoints[0].time:
384            return (self.waypoints[0].position, self.waypoints[0].velocity, self.waypoints[0].acceleration)
385        if t > self.waypoints[-1].time:
386            return (self.waypoints[-1].position, self.waypoints[-1].velocity, self.waypoints[-1].acceleration)
387
388        # Find segment indices
389        for i in range(self.get_num_segments()):
390            if t >= self.waypoints[i].time and t <= self.waypoints[i + 1].time:
391                w0 = self.waypoints[i]
392                w1 = self.waypoints[i + 1]
393                return hu.evaluate_polynomial_at(Segment.from_two_waypoints(w0, w1, segment_id=None).to_array(only_coeffs=True), t - w0.time)
394
395    def is_valid(self, v_des, a_des):
396        """Determines whether spline is well-formed and adheres to dynamic limits.
397
398        Parameters
399        ----------
400        v_des : float
401            Velocity limit that the spline shouldn't exceed
402        a_des : float
403            Acceleration limit that the spline shouldn't exceed
404
405        Returns
406        -------
407        Tuple(bool, str)
408            whether the segment is valid, and error message if not
409        """
410        # only check if valid if there are enough waypoints
411        if len(self.waypoints) < 2:
412            return True, "must have at least two waypoints"
413
414        # verify that spline starts at time zero
415        if not np.isclose(self.waypoints[0].time, 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
416            return False, "first waypoint must be planned for time zero"
417
418        # verify that waypoint time increases with index in the array
419        t = -1.0
420        for waypoint in self.waypoints:
421            if waypoint.time < 0.0:
422                return False, "waypoint cannot be planned for negative time"
423            if np.isclose(waypoint.time, t, atol=WAYPOINT_ISCLOSE_ATOL):
424                return False, "two waypoints cannot be planned for the same time"
425            if waypoint.time < t:
426                return False, "time must increase for each subsequent waypoint"
427            t = waypoint.time
428        # verify spline adheres to joint dynamics limits
429        for i in range(self.get_num_segments()):
430            success, v_max, a_max=hu.is_segment_feasible(self.get_segment(i).to_array(), v_des, a_des)
431            if not success:
432                return False, "segment %d exceeds dynamic bounds of (%f vel | %f acc ) with max of (%f vel | %f acc )"%(i,v_des,a_des,v_max,a_max)
433
434        return True, ""

577    def _compute_wheel_waypoints(self, translate_to_motor_rad, rotate_to_motor_rad,

582        # zero-th left and right waypoints

587                translate_to_motor_rad=translate_to_motor_rad, rotate_to_motor_rad=rotate_to_motor_rad)
588        if self.waypoints[0].accel_twist is not None:

594        right_wheel_waypoints.append(rw0)

596        # remaining left and right waypoints
597        for se2_w0, se2_w1 in zip(self.waypoints, self.waypoints[1:]):
598            lvel = rvel = None
599            laccel = raccel = None

658        """Determines whether trajectory is well-formed and adheres to dynamic limits.
659
660        Parameters
661        ----------
662        v_des : float
663            Velocity limit in motor space that the trajectory shouldn't exceed
664        a_des : float
665            Acceleration limit in motor space that the trajectory shouldn't exceed
666        translate_to_motor_rad : func or lambda
667            used to convert se2 waypoints into left motor space
668        rotate_to_motor_rad : func or lambda
669            used to convert se2 waypoints into right motor space
670
671        Returns
672        -------
673        Tuple(bool, str)
674            whether the segment is valid, and error message if not
675        """
676        # only check if valid if there are enough waypoints
677
678        if len(self.waypoints) < 2:
679            return True, "must have at least two waypoints"
680
681        # verify that spline starts at time zero
682        if not np.isclose(self.waypoints[0].time, 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
683            return False, "first waypoint must be planned for time zero"
684
685        # verify that spline starts at pose zero
686        if not np.isclose(self.waypoints[0].pose[0], 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
687            return False, "first base waypoint must have x component of pose be zero"
688        if not np.isclose(self.waypoints[0].pose[1], 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
689            return False, "first base waypoint must have y component of pose be zero"
690        if not np.isclose(self.waypoints[0].pose[2], 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
691            return False, "first base waypoint must have theta component of pose be zero"
692
693        # verify that waypoint time increases with index in the array
694        t = -1.0
695        for waypoint in self.waypoints:
696            if waypoint.time < 0.0:
697                return False, "waypoint cannot be planned for negative time"
698            if np.isclose(waypoint.time, t, atol=WAYPOINT_ISCLOSE_ATOL):
699                return False, "two waypoints cannot be planned for the same time"
700            if waypoint.time < t:
701                return False, "time must increase for each subsequent waypoint"
702            t = waypoint.time
703
704        # verify left and right trajectories adheres to joint dynamics limits
705        for i in range(self.get_num_segments()):
706            ls, rs = self.get_wheel_segments(i, translate_to_motor_rad, rotate_to_motor_rad)
707            success, v_max, a_max =hu.is_segment_feasible(ls.to_array(), v_des, a_des)
708            if not success:
709                return False, "left wheel segment %d exceeds dynamic bounds of (%f vel | %f acc ) with max of (%f vel | %f acc )"%(i,v_des,a_des,v_max,a_max)
710            success, v_max, a_max =hu.is_segment_feasible(rs.to_array(), v_des, a_des)
711            if not success:
712                return False, "right wheel segment %d exceeds dynamic bounds of (%f vel | %f acc ) with max of (%f vel | %f acc )"%(i,v_des,a_des,v_max,a_max)
713
714        # verify that either translate or rotate only at a time
715        for i in range(1,len(self.waypoints)):
716            dx=self.waypoints[i].pose[0]-self.waypoints[i-1].pose[0]
717            dy = self.waypoints[i].pose[1] - self.waypoints[i-1].pose[1]
718            dtheta = self.waypoints[i].pose[2] - self.waypoints[i-1].pose[2]
719            translating = (not np.isclose(dx, 0.0, atol=WAYPOINT_ISCLOSE_ATOL)) or (not np.isclose(dy, 0.0, atol=WAYPOINT_ISCLOSE_ATOL))
720            rotating = (not np.isclose(dtheta, 0.0, atol=WAYPOINT_ISCLOSE_ATOL))
721            if translating and rotating:
722                return False, 'DiffDriveTrajectory waypoint cannot both translate and rotate: \n%s'%str(self.waypoints[i])
723
724        return True, ""

302    def add_waypoint(self, new_waypoint):
303        """Add a waypoint to the spline.
304
305        This method will sort through the existing waypoints
306        in the spline to insert the waypoint such that
307        waypoint time increases with index in the array.
308
309        Parameters
310        ----------
311        waypoint : ``Waypoint``
312            with time, position, [velocity, [and acceleration]] attributes set
313        """
314        if len(self.waypoints) == 0:
315            self.waypoints.append(new_waypoint)
316            return
317
318        # Cannot have two waypoints scheduled for the same time
319        if new_waypoint in self.waypoints:
320            return
321
322        # Prepend or append if before first or after last waypoint
323        if new_waypoint < self.waypoints[0]:
324            self.waypoints.insert(0, new_waypoint)
325            return
326        if new_waypoint > self.waypoints[0]:
327            self.waypoints.append(new_waypoint)
328            return
329
330        # Insert before first later waypoint
331        for i, other_waypoint in enumerate(self.waypoints):
332            if new_waypoint < other_waypoint:
333                self.waypoints.insert(i, new_waypoint)
334                return

365    def evaluate_at(self, t, to_motor_rad=lambda pos: pos):
366        """Evaluate a point along the curve at a given time.
367
368        Parameters
369        ----------
370        t : float
371            time in seconds
372        to_motor_rad : func or lambda
373            optional, used to convert waypoint into motor space
374
375        Returns
376        -------
377        Tuple(float)
378            tuple with three elements: evaluated position, velocity, and acceleration.
379        """
380        if len(self.waypoints) < 2:
381            return
382
383        # Return bounds for early or late t
384        if t < self.waypoints[0].time:
385            return (self.waypoints[0].position, self.waypoints[0].velocity, self.waypoints[0].acceleration)
386        if t > self.waypoints[-1].time:
387            return (self.waypoints[-1].position, self.waypoints[-1].velocity, self.waypoints[-1].acceleration)
388
389        # Find segment indices
390        for i in range(self.get_num_segments()):
391            if t >= self.waypoints[i].time and t <= self.waypoints[i + 1].time:
392                w0 = self.waypoints[i]
393                w1 = self.waypoints[i + 1]
394                return hu.evaluate_polynomial_at(Segment.from_two_waypoints(w0, w1, segment_id=None).to_array(only_coeffs=True), t - w0.time)

396    def is_valid(self, v_des, a_des):
397        """Determines whether spline is well-formed and adheres to dynamic limits.
398
399        Parameters
400        ----------
401        v_des : float
402            Velocity limit that the spline shouldn't exceed
403        a_des : float
404            Acceleration limit that the spline shouldn't exceed
405
406        Returns
407        -------
408        Tuple(bool, str)
409            whether the segment is valid, and error message if not
410        """
411        # only check if valid if there are enough waypoints
412        if len(self.waypoints) < 2:
413            return True, "must have at least two waypoints"
414
415        # verify that spline starts at time zero
416        if not np.isclose(self.waypoints[0].time, 0.0, atol=WAYPOINT_ISCLOSE_ATOL):
417            return False, "first waypoint must be planned for time zero"
418
419        # verify that waypoint time increases with index in the array
420        t = -1.0
421        for waypoint in self.waypoints:
422            if waypoint.time < 0.0:
423                return False, "waypoint cannot be planned for negative time"
424            if np.isclose(waypoint.time, t, atol=WAYPOINT_ISCLOSE_ATOL):
425                return False, "two waypoints cannot be planned for the same time"
426            if waypoint.time < t:
427                return False, "time must increase for each subsequent waypoint"
428            t = waypoint.time
429        # verify spline adheres to joint dynamics limits
430        for i in range(self.get_num_segments()):
431            success, v_max, a_max=hu.is_segment_feasible(self.get_segment(i).to_array(), v_des, a_des)
432            if not success:
433                return False, "segment %d exceeds dynamic bounds of (%f vel | %f acc ) with max of (%f vel | %f acc )"%(i,v_des,a_des,v_max,a_max)
434
435        return True, ""

692            return False, "first base waypoint must have theta component of pose be zero"

703            t = waypoint.time

716        for i in range(1,len(self.waypoints)):
717            dx=self.waypoints[i].pose[0]-self.waypoints[i-1].pose[0]
718            dy = self.waypoints[i].pose[1] - self.waypoints[i-1].pose[1]
